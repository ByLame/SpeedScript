-- ==================== DRIVE FARM FUNCTIONS ====================
local function fastDriveSpam()
    local driveNames = {"Red Drive", "White Drive", "Blue Drive", "Glitched Drive"}
    for _ = 1, 5 do 
        for _, name in ipairs(driveNames) do
            task.spawn(function()
                functions.PlayerActivesCommand:InvokeServer({ Name = name })
            end)
        end
        task.wait() 
    end
end

local function getInventoryCount(driveName)
    local screenGui = player.PlayerGui:FindFirstChild("ScreenGui")
    local eggRows = screenGui and screenGui:FindFirstChild("EggRows", true)
    if not eggRows then return 0 end
    
    for _, row in ipairs(eggRows:GetChildren()) do
        local typeLabel = row:FindFirstChild("TypeName")
        if typeLabel and typeLabel.Text == driveName then
            local countText = row:FindFirstChild("EggSlot") and row.EggSlot:FindFirstChild("Count") and row.EggSlot.Count.Text
            return tonumber(countText:match("(%d+)/")) or 0
        end
    end
    return 0
end

local function checkDigitalBee()
    local prompt = player.PlayerGui.ScreenGui:FindFirstChild("RoboBearPrompt")
    if not (prompt and prompt.Visible) then return nil end
    local btnFrame = prompt:FindFirstChild("ButtonFrame", true)
    for i = 1, 3 do
        local btn = btnFrame and btnFrame:FindFirstChild("Button" .. i)
        if btn and btn:FindFirstChild("Text1") and string.find(btn.Text1.Text, "★ Digital Bee") then
            return i
        end
    end
    return nil
end

local function startAutomation()
    while config.active do
        -- 1. HYPER RESTOCK
        for displayName, technicalName in pairs(driveTargets) do
            while getInventoryCount(displayName) < 5 and config.active do
                remotes.SelectNPCOption:FireServer("StartRoboBearChallenge")
                remotes.roboBearChallenge:FireServer("start")
                remotes.roboBearChallenge:FireServer("quit")
                
                task.spawn(function()
                    functions.ItemPackageEvent:InvokeServer("Purchase", {
                        ["Type"] = technicalName,
                        ["Category"] = "Eggs",
                        ["Amount"] = 1
                    })
                end)
                
                task.wait(0.01)
            end
        end

        -- 2. OPTIMIZED SNIPE
        if config.active then
            remotes.SelectNPCOption:FireServer("StartRoboBearChallenge")
            task.wait()
            remotes.roboBearChallenge:FireServer("selectQuest", 1)
            task.wait()

            local found = false
            for i = 0, 6 do
                local beeIndex = checkDigitalBee()
                if beeIndex then
                    remotes.roboBearChallenge:FireServer("selectBee", beeIndex)
                    remotes.roboBearChallenge:FireServer("start")
                    
                    task.wait(0.01)
                    fastDriveSpam() 
                    
                    task.wait(0.3)
                    remotes.roboBearChallenge:FireServer("quit")
                    found = true
                    break
                end
                
                if i < 6 then
                    remotes.roboBearChallenge:FireServer("reroll")
                    task.wait(0.1) 
                end
            end
            if not found then remotes.roboBearChallenge:FireServer("quit") end
        end
        task.wait(0.01)
    end
end

-- ==================== SPEED SYSTEM ====================
local function applySpeed(character)
    local humanoid = character:WaitForChild("Humanoid")
    local rootPart = character:WaitForChild("HumanoidRootPart")

    for _, part in pairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            part.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.3, 0, 0.1, 0)
        end
    end

    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not humanoid or not humanoid.Parent then
            connection:Disconnect()
            return
        end
        if config.speedEnabled then
            humanoid.WalkSpeed = config.walkSpeed
        else
            humanoid.WalkSpeed = 16
        end
        if humanoid.MoveDirection.Magnitude == 0 then
            rootPart.AssemblyLinearVelocity = Vector3.new(0, rootPart.AssemblyLinearVelocity.Y, 0)
        end
    end)

    humanoid.Died:Connect(function()
        connection:Disconnect()
    end)
end

if player.Character then
    applySpeed(player.Character)
end

player.CharacterAdded:Connect(function(character)
    task.wait(0.5)
    applySpeed(character)
end)
