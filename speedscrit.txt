-- Speed Logic
local enabled = true
local STEP = 10 -- how much +/- changes speed

local function updateSpeedLabel()
    speedLabel.Text = "Speed: " .. getgenv().WalkSpeed
    speedInput.Text = tostring(getgenv().WalkSpeed)
end

-- Plus / Minus buttons
plusBtn.MouseButton1Click:Connect(function()
    getgenv().WalkSpeed = (getgenv().WalkSpeed or 120) + STEP
    updateSpeedLabel()
end)

minusBtn.MouseButton1Click:Connect(function()
    getgenv().WalkSpeed = math.max(16, (getgenv().WalkSpeed or 120) - STEP)
    updateSpeedLabel()
end)

-- Manual input box
speedInput.FocusLost:Connect(function()
    local val = tonumber(speedInput.Text)
    if val then
        getgenv().WalkSpeed = math.max(16, val)
    end
    updateSpeedLabel()
end)

-- Toggle button
toggleBtn.MouseButton1Click:Connect(function()
    enabled = not enabled
    if enabled then
        toggleBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 0)
        toggleBtn.Text = "Speed: ON"
    else
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
        toggleBtn.Text = "Speed: OFF"
    end
end)

-- Right/Left arrow to hide/show
UserInputService.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.Right then
        frame.Visible = false
    elseif input.KeyCode == Enum.KeyCode.Left then
        frame.Visible = true
    end
end)

local function applySpeed(character)
    local humanoid = character:WaitForChild("Humanoid")
    local rootPart = character:WaitForChild("HumanoidRootPart")

    for _, part in pairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            part.CustomPhysicalProperties = PhysicalProperties.new(
                0.7, 0.3, 0, 0.1, 0
            )
        end
    end

    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not humanoid or not humanoid.Parent then
            connection:Disconnect()
            return
        end

        if enabled then
            humanoid.WalkSpeed = getgenv().WalkSpeed or 120
        else
            humanoid.WalkSpeed = 16
        end

        if humanoid.MoveDirection.Magnitude == 0 then
            rootPart.AssemblyLinearVelocity = Vector3.new(
                0,
                rootPart.AssemblyLinearVelocity.Y,
                0
            )
        end
    end)

    humanoid.Died:Connect(function()
        connection:Disconnect()
    end)
end

if player.Character then
    applySpeed(player.Character)
end

player.CharacterAdded:Connect(function(character)
    task.wait(0.5)
    applySpeed(character)
end)
