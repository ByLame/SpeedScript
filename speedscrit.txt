Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 10)

-- Dragging
local dragging, dragStart, startPos
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
    end
end)
frame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

-- Update label helper
local function updateLabel()
    speedLabel.Text = "Speed: " .. getgenv().WalkSpeed
    speedInput.Text = tostring(getgenv().WalkSpeed)
end

-- Plus / Minus
plusBtn.MouseButton1Click:Connect(function()
    getgenv().WalkSpeed = getgenv().WalkSpeed + 10
    updateLabel()
end)

minusBtn.MouseButton1Click:Connect(function()
    getgenv().WalkSpeed = math.max(16, getgenv().WalkSpeed - 10)
    updateLabel()
end)

-- Manual input
speedInput.FocusLost:Connect(function(enterPressed)
    local val = tonumber(speedInput.Text)
    if val and val >= 16 then
        getgenv().WalkSpeed = val
    end
    updateLabel()
end)

-- Toggle
toggleBtn.MouseButton1Click:Connect(function()
    getgenv().SpeedEnabled = not getgenv().SpeedEnabled
    if getgenv().SpeedEnabled then
        toggleBtn.BackgroundColor3 = Color3.fromRGB(0, 180, 0)
        toggleBtn.Text = "Speed: ON"
    else
        toggleBtn.BackgroundColor3 = Color3.fromRGB(180, 0, 0)
        toggleBtn.Text = "Speed: OFF"
    end
end)

-- Arrow keys hide/show
UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if input.KeyCode == Enum.KeyCode.Right then
        frame.Visible = false
    elseif input.KeyCode == Enum.KeyCode.Left then
        frame.Visible = true
    end
end)

-- Speed apply
local function applySpeed(character)
    local humanoid = character:WaitForChild("Humanoid")
    local rootPart = character:WaitForChild("HumanoidRootPart")

    for _, part in pairs(character:GetChildren()) do
        if part:IsA("BasePart") then
            part.CustomPhysicalProperties = PhysicalProperties.new(0.7, 0.3, 0, 0.1, 0)
        end
    end

    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not humanoid or not humanoid.Parent then
            connection:Disconnect()
            return
        end
        if getgenv().SpeedEnabled then
            humanoid.WalkSpeed = getgenv().WalkSpeed
        else
            humanoid.WalkSpeed = 16
        end
        if humanoid.MoveDirection.Magnitude == 0 then
            rootPart.AssemblyLinearVelocity = Vector3.new(0, rootPart.AssemblyLinearVelocity.Y, 0)
        end
    end)

    humanoid.Died:Connect(function()
        connection:Disconnect()
    end)
end

if player.Character then
    applySpeed(player.Character)
end

player.CharacterAdded:Connect(function(character)
    task.wait(0.5)
    applySpeed(character)
end)
